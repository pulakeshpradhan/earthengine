    গুগল আর্থ ইঞ্জিনে চার্টসমূহ (বিস্তারিত)¶

    ১. ইমেজ-ভিত্তিক চার্ট¶

    টাইম সিরিজ বিশ্লেষণ¶

    • ui.Chart.image.series: সময়ের সাথে একক অঞ্চল
    • ui.Chart.image.seriesByRegion: সময়ের সাথে একাধিক অঞ্চল
    • ui.Chart.image.doySeries: একক অঞ্চলের বছরের-দিন প্যাটার্ন
    • ui.Chart.image.doySeriesByYear: একক অঞ্চলের জন্য বছর তুলনা
    • ui.Chart.image.doySeriesByRegion: অঞ্চল জুড়ে বছরের-দিন প্যাটার্ন

    স্থানিক বিশ্লেষণ¶

    • ui.Chart.image.byRegion: একাধিক অঞ্চল জুড়ে একক-ব্যান্ড সারসংক্ষেপ
    • ui.Chart.image.regions: একাধিক অঞ্চল জুড়ে টাইম সিরিজ
    • ui.Chart.image.byClass: অঞ্চল শ্রেণী অনুসারে ব্যান্ড বিতরণ
    • ui.Chart.image.histogram: পিক্সেল মান বিতরণ

    ২. ফিচার-ভিত্তিক চার্ট¶

    এই ফাংশনগুলো FeatureCollection-এর সাথে কাজ করে।

    প্রপার্টি বিশ্লেষণ¶

    • ui.Chart.feature.byFeature: পৃথক ফিচার প্রপার্টি
    • ui.Chart.feature.byProperty: একাধিক প্রপার্টি তুলনা
    • ui.Chart.feature.groups: গ্রুপভিত্তিক ডেটা তুলনা
    • ui.Chart.feature.histogram: প্রপার্টি মান বিতরণ

    সাধারণ প্যারামিটার¶

    • scale: মিটারে স্থানিক রেজোলিউশন
    • reducer: পিক্সেল মান সমষ্টিকরণ ফাংশন
    • region/regions: বিশ্লেষণের জন্য আগ্রহের এলাকা
    • xProperty: x-অক্ষ মানের প্রপার্টি
    • seriesProperty: একাধিক সিরিজ আলাদা করার প্রপার্টি

    ১. ui.Chart.image.series¶

    বর্ণনা¶

    নির্দিষ্ট অঞ্চলে সময়ের সাথে গাছপালা সূচক (NDVI ও EVI) পরিবর্তন দেখায়।

    var forest = ee.FeatureCollection('projects/google/charts_feature_example')
                 .filter(ee.Filter.eq('label', 'Forest'));
var vegIndices = ee.ImageCollection('MODIS/061/MOD13A1')
                     .filter(ee.Filter.date('2010-01-01', '2020-01-01'))
                     .select(['NDVI', 'EVI']);

var chart = ui.Chart.image.series({
  imageCollection: vegIndices, region: forest,
  reducer: ee.Reducer.mean(), scale: 500, xProperty: 'system:time_start'
}).setOptions({title: 'বনের জন্য তারিখ অনুসারে গড় গাছপালা সূচক মান'});
print(chart);

    ২. ui.Chart.image.seriesByRegion¶

    বর্ণনা¶

    একাধিক অঞ্চলে সময়ের সাথে NDVI মান তুলনা করে।

    var ecoregions = ee.FeatureCollection('projects/google/charts_feature_example');
var chart = ui.Chart.image.seriesByRegion({
  imageCollection: vegIndices, band: 'NDVI', regions: ecoregions,
  reducer: ee.Reducer.mean(), scale: 500,
  seriesProperty: 'label', xProperty: 'system:time_start'
}).setOptions({title: 'তারিখ অনুসারে গড় NDVI মান'});
print(chart);

    ৩. ui.Chart.image.doySeries¶

    বর্ণনা¶

    একাধিক বছর জুড়ে বছরের প্রতিটি দিনে গড় NDVI ও EVI প্লট করে। মৌসুমী প্রবণতা চিহ্নিতকরণে সাহায্য করে।

    var grassland = ee.FeatureCollection('projects/google/charts_feature_example')
                    .filter(ee.Filter.eq('label', 'Grassland'));
var chart = ui.Chart.image.doySeries({
  imageCollection: vegIndices, region: grassland,
  regionReducer: ee.Reducer.mean(), scale: 500,
  yearReducer: ee.Reducer.mean(), startDay: 1, endDay: 365
}).setOptions({title: 'তৃণভূমির বছরের দিন অনুসারে গড় গাছপালা সূচক'});
print(chart);

    ৪. ui.Chart.image.doySeriesByYear¶

    বর্ণনা¶

    বিভিন্ন বছরে নির্দিষ্ট অঞ্চলের NDVI প্রবণতা তুলনা করে।

    var chart = ui.Chart.image.doySeriesByYear({
  imageCollection: vegIndices, bandName: 'NDVI', region: grassland,
  regionReducer: ee.Reducer.mean(), scale: 500,
  sameDayReducer: ee.Reducer.mean(), startDay: 1, endDay: 365
}).setOptions({title: 'তৃণভূমির বছরের দিন অনুসারে গড় NDVI মান'});
print(chart);

    ৫. ui.Chart.image.doySeriesByRegion¶

    বর্ণনা¶

    একটি বছরে বিভিন্ন অঞ্চলে গড় NDVI মান তুলনা করে।

    var chart = ui.Chart.image.doySeriesByRegion({
  imageCollection: vegIndices, bandName: 'NDVI', regions: ecoregions,
  regionReducer: ee.Reducer.mean(), scale: 500,
  yearReducer: ee.Reducer.mean(), seriesProperty: 'label',
  startDay: 1, endDay: 365
}).setOptions({title: 'বছরের দিন অনুসারে গড় NDVI মান'});
print(chart);

    ৬. ui.Chart.image.byRegion¶

    উদ্দেশ্য¶

    একটি ইমেজের একক ব্যান্ডের মান একাধিক অঞ্চল জুড়ে সংক্ষিপ্ত করে।

    var modisImage = ee.Image('MODIS/061/MOD13A1/2010_01_01').select('NDVI');
var chart = ui.Chart.image.byRegion({
  image: modisImage, regions: ecoregions,
  reducer: ee.Reducer.mean(), scale: 500, xProperty: 'label'
}).setOptions({title: 'ইকোরিজিয়নের গড় NDVI'});
print(chart);

    ৭. ui.Chart.image.regions¶

    উদ্দেশ্য¶

    একটি অঞ্চলের জন্য এক বা একাধিক ব্যান্ডের মান ব্যবহার করে টাইম সিরিজ প্লট করে।

    var meanNDVI = vegIndices.mean();
var chart = ui.Chart.image.regions({
  image: meanNDVI, regions: grassland,
  reducer: ee.Reducer.mean(), scale: 500, seriesProperty: 'label'
}).setOptions({title: 'তৃণভূমির NDVI টাইম সিরিজ'});
print(chart);

    ৮. ui.Chart.image.byClass¶

    উদ্দেশ্য¶

    একটি অঞ্চলে সংজ্ঞায়িত বিভিন্ন শ্রেণী জুড়ে ইমেজ ব্যান্ডের মান বিতরণ তুলনা করে।

    var chart = ui.Chart.image.byClass({
  image: modisImage, regions: ecoregions,
  classProperty: 'label', reducer: ee.Reducer.mean(), scale: 500
}).setOptions({title: 'শ্রেণী অনুসারে NDVI বিতরণ'});
print(chart);

    ৯. ui.Chart.image.histogram¶

    উদ্দেশ্য¶

    পিক্সেল মানের উপর ভিত্তি করে ইমেজের ব্যান্ডের হিস্টোগ্রাম তৈরি করে।

    var chart = ui.Chart.image.histogram({
  image: modisImage, region: grassland, scale: 500, maxBuckets: 20
}).setOptions({title: 'তৃণভূমির NDVI হিস্টোগ্রাম'});
print(chart);

    ১০. ui.Chart.feature.byFeature¶

    উদ্দেশ্য¶

    FeatureCollection-এ প্রতিটি ফিচারের প্রপার্টি প্রদর্শন করে।

    var chart = ui.Chart.feature.byFeature({
  features: ecoregionsWithNDVI, xProperty: 'label', yProperties: ['mean']
}).setOptions({title: 'অঞ্চল অনুসারে গড় NDVI'});
print(chart);

    ১১. ui.Chart.feature.byProperty¶

    উদ্দেশ্য¶

    FeatureCollection-এর একাধিক সংখ্যাসূচক প্রপার্টি সংক্ষিপ্ত করে।

    var chart = ui.Chart.feature.byProperty({
  features: ecoregionsWithVeg, xProperties: ['NDVI', 'EVI']
}).setOptions({title: 'প্রপার্টি অনুসারে গাছপালা সূচক তুলনা'});
print(chart);

    ১২. ui.Chart.feature.groups¶

    উদ্দেশ্য¶

    শ্রেণীভুক্ত প্রপার্টির উপর ভিত্তি করে গ্রুপভিত্তিক ডেটা তুলনা করে।

    var chart = ui.Chart.feature.groups({
  features: ecoregionsWithGroups, xProperty: 'group', yProperty: 'mean'
}).setOptions({title: 'গ্রুপভিত্তিক গড় NDVI মান'});
print(chart);

    ১৩. ui.Chart.feature.histogram¶

    উদ্দেশ্য¶

    FeatureCollection-এর ফিচার জুড়ে সংখ্যাসূচক প্রপার্টির হিস্টোগ্রাম তৈরি করে।

    var chart = ui.Chart.feature.histogram({
  features: ecoregionsWithNDVI, property: 'mean', minBucketWidth: 0.1
}).setOptions({title: 'NDVI মানের হিস্টোগ্রাম'});
print(chart);
