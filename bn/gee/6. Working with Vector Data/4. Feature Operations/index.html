    Feature Operations

    ফিচার বনাম ফিচার কালেকশন¶

    • ফিচার (Feature): সংশ্লিষ্ট বৈশিষ্ট্য (প্রপার্টি) সহ একটি একক ভৌগোলিক বস্তু। এটি একটি জ্যামিতি (যেমন- পয়েন্ট, লাইন, পলিগন) এবং মেটাডেটা নিয়ে গঠিত।
    • ফিচার কালেকশন (FeatureCollection): একাধিক ফিচারের সংকলন। এটি একাধিক ভৌগোলিক বস্তু ধারণকারী ডেটাসেট সংরক্ষণ ও প্রক্রিয়াকরণে ব্যবহৃত হয়।

    FeatureCollection ইম্পোর্ট করা¶

    var table = ee.FeatureCollection("projects/pulakesh/assets/cmc");
var city = ee.FeatureCollection("projects/pulakesh/assets/bmc");
Map.addLayer(table);
    • ee.FeatureCollection: ইম্পোর্ট করা স্থানিক ডেটা প্রতিনিধিত্ব করে।
    • Map.addLayer: ভিজ্যুয়ালাইজেশনের জন্য ডেটা মানচিত্রে যোগ করে।

    ১. ক্ষেত্রফল গণনা¶

    print(table.geometry().area().divide(1e6));
    • উদ্দেশ্য: সম্পূর্ণ FeatureCollection-এর ক্ষেত্রফল বর্গ কিলোমিটারে গণনা করে।
    var area = table.map(function(feature) {
  return feature.set('area_km2', feature.geometry().area().divide(1e6));
});
    • প্রতিটি ফিচারে area_km2 প্রপার্টি যোগ করে।

    ২. সীমানা (Bounds)¶

    var bounds = table.map(function(feature) {
  return feature.setGeometry(feature.geometry().bounds());
});
    • উদ্দেশ্য: প্রতিটি ফিচারের বাউন্ডিং বক্স গণনা করে।

    ৩. বাফার (Buffer)¶

    var buffered = table.map(function(feature) {
  return feature.buffer(10000); // মিটারে বাফার দূরত্ব
});
    • উদ্দেশ্য: প্রতিটি ফিচারের চারপাশে ১০ কিমি বাফার জোন তৈরি করে।

    ৪. কেন্দ্রবিন্দু (Centroid)¶

    var centroids = table.map(function(feature) {
  return feature.setGeometry(feature.geometry().centroid());
});
    • উদ্দেশ্য: প্রতিটি ফিচারের জ্যামিতিক কেন্দ্র গণনা করে।

    ৫. দূরত্ব (Distance)¶

    var point = ee.Geometry.Point([37.5, 0.5]);
var distances = table.map(function(feature) {
  return feature.set('distance_to_point_m', feature.geometry().distance(point));
});
    • উদ্দেশ্য: প্রতিটি ফিচারের একটি নির্দিষ্ট পয়েন্ট থেকে দূরত্ব গণনা করে।

    ৬. ডিজলভ (Dissolve)¶

    var dissolved = table.geometry();
    • উদ্দেশ্য: সব ফিচারকে একটি একক জ্যামিতিতে একত্রিত করে।

    ৭. মূল্যায়ন (Evaluate)¶

    table.first().geometry().area().evaluate(function(area) {
  print('প্রথম ফিচারের ক্ষেত্রফল (বর্গমি.):', area);
});
    • উদ্দেশ্য: প্রথম ফিচারের ক্ষেত্রফল ক্লায়েন্ট-সাইডে মূল্যায়ন করে।

    ৮. জ্যামিতি (Geometry)¶

    var firstGeometry = table.first().geometry();

    ৯. প্রপার্টি নাম¶

    var propertyNames = table.first().propertyNames();

    ১০. নির্বাচন (Select)¶

    var selected = table.select(['ADM1_NAME', 'ADM0_NAME']);

    ১১. সরলীকরণ (Simplify)¶

    var simplified = table.map(function(feature) {
  return feature.setGeometry(feature.geometry().simplify(1000));
});

    ১২. ইউনিয়ন (Union)¶

    var fc = ee.FeatureCollection('WRI/GPPD/power_plants')
            .filter('country_lg == "Belgium"');
var union = fc.union(1);

    ১৩. দূরত্বের মধ্যে (Within Distance)¶

    var withinDistance = table.filter(ee.Filter.withinDistance({
  distance: 50000, leftField: '.geo', rightField: '.geo', maxError: 10
}));

    ১৪. অতিরিক্ত অপারেশন¶

    • ছেদ (Intersection):
    var intersection = table.geometry().intersection(table.geometry());
    • পার্থক্য (Difference):
    var difference = table.geometry().difference(buffered.geometry());

    ১৫. এক্সপোর্ট¶

    Export.table.toDrive({
  collection: area, description: 'Exported_Features', fileFormat: 'CSV'
});
