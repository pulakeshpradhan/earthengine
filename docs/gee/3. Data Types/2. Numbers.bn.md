```javascript
// সংখ্যামূলক ভ্যারিয়েবল (Numeric Variable)
var num = ee.Number(10);
print('Numeric:', num);  // প্রিন্ট করবে 10

// গাণিতিক অপারেটর (Mathematical Operators)
var a = ee.Number(5);
var b = ee.Number(2);
print('Add (যোগ):', a.add(b));  // প্রিন্ট করবে 7
print('Subtract (বিয়োগ):', a.subtract(b));  // প্রিন্ট করবে 3
print('Multiply (গুণ):', a.multiply(b));  // প্রিন্ট করবে 10
print('Divide (ভাগ):', a.divide(b));  // প্রিন্ট করবে 2.5

// স্ট্রিং বা টেক্সট ভ্যারিয়েবল
var str = ee.String('Hello, world!');
print('String:', str);  // প্রিন্ট করবে 'Hello, world!'

// বুলিয়ান ভ্যারিয়েবল (সত্য/মিথ্যা)
var bool = true;
print('Boolean:', bool);  // প্রিন্ট করবে true

// লিস্ট বা তালিকা
var list = ee.List([1, 2, 3, 4, 5]);
print('List:', list);  // প্রিন্ট করবে [1, 2, 3, 4, 5]

// ডিকশনারি
var dict = ee.Dictionary({name: 'John', age: 30, city: 'New York'});
print('Dictionary:', dict);  // প্রিন্ট করবে {name: 'John', age: 30, city: 'New York'}
```
