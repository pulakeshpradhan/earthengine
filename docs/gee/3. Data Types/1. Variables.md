```javascript
// Numerical Variable
var num = ee.Number(10);
print('Numeric:', num);  // Prints 10 

// String Variable
var str = ee.String('Hello, world!');
print('String:', str);  // Prints 'Hello, world!'

// List (Array) Variable
var list = ee.List([1, 2, 3, 4, 5]);
print('List:', list);  // Prints [1, 2, 3, 4, 5]

// Dictionary (Object) Variable
var dict = ee.Dictionary({name: 'John', age: 30, city: 'New York'});
print('Dictionary:', dict);  // Prints {name: 'John', age: 30, city: 'New York'}

// Date Variable
var date = ee.Date('2022-01-01');
print('Date:', date);  // Prints '2022-01-01'

// Array Variable
var array = ee.Array([[1, 2, 3], [4, 5, 6]]);
print('Array:', array);
