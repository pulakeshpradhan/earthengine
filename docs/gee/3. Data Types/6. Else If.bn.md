```javascript
// ভ্যারিয়েবল ডিফাইন করুন
var x = ee.Number(10);
var y = ee.Number(10);
var z = ee.Number(10);

// 'x > 5' এর জন্য If কন্ডিশন
var resultX = ee.Algorithms.If(x.gt(5), 'x ৫-এর চেয়ে বড়', 'x ৫-এর চেয়ে বড় নয়');
print(resultX); 

// 'y > 15' অথবা 'y > 5' এর জন্য Else if এর সমতুল্য কোড
var resultY = ee.Algorithms.If(
    y.gt(15), 
    'y ১৫-এর চেয়ে বড়', 
    ee.Algorithms.If(y.gt(5), 'y ৫-এর চেয়ে বড়', 'y ৫ অথবা তার কম')
);
print(resultY); 

// 'z > 15' অথবা 'else' এর জন্য
var resultZ = ee.Algorithms.If(
    z.gt(15), 
    'z ১৫-এর চেয়ে বড়', 
    'z ১৫-এর চেয়ে বড় নয়'
);
print(resultZ); 
```
