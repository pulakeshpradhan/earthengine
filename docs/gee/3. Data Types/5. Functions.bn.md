```javascript
// সাধারণ ফাংশন
function greet() {
    print('Hello, world!');
}
greet();  // ফাংশনটি কল করা হলো 

// যোগ করার ফাংশন
function sum(a, b) {
    return a + b;
}
print('Sum (যোগফল):', sum(5, 3));  // প্রিন্ট করবে 8

// ইনডেক্স খুঁজে বের করার ফাংশন
function getIndex(array, element) {
    return array.indexOf(element);
}
print('Index (অবস্থান):', getIndex([1, 2, 3, 4, 5], 3));  // প্রিন্ট করবে 2

// গড় নির্ণয়ের ফাংশন
function average(array) {
    var sum = array.reduce(function(a, b) { return a + b; }, 0);
    return sum / array.length;
}
print('Average (গড়):', average([1, 2, 3, 4, 5]));  // প্রিন্ট করবে 3

// বর্গ নির্ণয়ের ফাংশন
function square(x) {
    return x * x;
}
print('Square (বর্গ):', square(5));  // প্রিন্ট করবে 25
```
