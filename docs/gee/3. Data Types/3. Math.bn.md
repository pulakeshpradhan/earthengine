```javascript
// পাই (PI) এবং ই (E)-এর মান
var pi = ee.Number(Math.PI);
print('PI:', pi);
var e = ee.Number(Math.E);
print('E:', e);

// সংখ্যা রাউন্ড করা (ceil, trunc, round) 
var x = ee.Number(10.3);
var ceilValue = x.ceil();  // ১১ (পরবর্তী পূর্ণসংখ্যা)
print('Ceil:', ceilValue);
var truncValue = x.floor();  // ১০ (আগের পূর্ণসংখ্যা)
print('Trunc (Floor):', truncValue);
var roundValue = x.round();  // ১০ (নিকটবর্তী পূর্ণসংখ্যা)
print('Round:', roundValue);

// এক্সপোনেন্ট (Exponent)
var expValue = e.exp(); 
print('Exp:', expValue);

// লগারিদম (Logarithm)
var logValue = x.log(); 
print('Log:', logValue);

// পাওয়ার বা ঘাত (Power)
var y = ee.Number(3);
var powValue = x.pow(y); 
print('Pow:', powValue);

// বর্গমূল (Square root)
var sqrtValue = x.sqrt(); 
print('Sqrt:', sqrtValue);

// পরম মান (Absolute value)
var absValue = x.abs(); 
print('Abs:', absValue);

// ত্রিকোণমিতি ফাংশন (Trigonometric functions)
var angleInRadians = ee.Number(Math.PI / 4); // ৪৫ ডিগ্রি রেডিয়ানে
var sinValue = angleInRadians.sin();
print('Sin:', sinValue);
var cosValue = angleInRadians.cos();
print('Cos:', cosValue);
var tanValue = angleInRadians.tan();
print('Tan:', tanValue);
```
