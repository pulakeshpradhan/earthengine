```javascript
// Numeric Variable
var num = ee.Number(10);
print('Numeric:', num);  // Prints 10

// Mathematical Operators 
var a = ee.Number(5);
var b = ee.Number(2);
print('Add:', a.add(b));  // Prints 7
print('Subtract:', a.subtract(b));  // Prints 3
print('Multiply:', a.multiply(b));  // Prints 10
print('Divide:', a.divide(b));  // Prints 2.5

// String Variable
var str = ee.String('Hello, world!');
print('String:', str);  // Prints 'Hello, world!'

// Boolean Variable
var bool = true;
print('Boolean:', bool);  // Prints true

// List (Array)
var list = ee.List([1, 2, 3, 4, 5]);
print('List:', list);  // Prints [1, 2, 3, 4, 5]

// Dictionary (Object)
var dict = ee.Dictionary({name: 'John', age: 30, city: 'New York'});
print('Dictionary:', dict);  // Prints {name: 'John', age: 30, city: 'New York'}
