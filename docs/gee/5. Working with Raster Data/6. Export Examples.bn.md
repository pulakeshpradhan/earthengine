### এক্সপোর্ট কোড

```javascript
// ধাপ ৫: র‍্যাস্টার ইমেজ এক্সপোর্ট করুন
Export.image.toDrive({
  image: clippedRaster,           // এক্সপোর্ট করতে চাওয়া ইমেজ
  description: 'Exported_Raster', // এক্সপোর্ট টাস্কের নাম
  folder: 'GEE_Exports',          // গুগল ড্রাইভে ফোল্ডার
  fileNamePrefix: 'raster_example',// ফাইলের নাম
  region: aoi,                     // এক্সপোর্ট করতে চাওয়া এলাকা
  scale: 30,                       // রেজোলিউশন (প্রতি পিক্সেল ৩০মি)
  crs: 'EPSG:4326',               // কোঅর্ডিনেট রেফারেন্স সিস্টেম
  maxPixels: 1e13                  // সর্বোচ্চ পিক্সেল সংখ্যা
});

// ধাপ ৬: ভেক্টর ডেটা AOI-তে ফিল্টার করুন
var filteredVector = vector.filterBounds(aoi);
Map.addLayer(filteredVector, {}, 'ফিল্টার করা ভেক্টর');

// ধাপ ৭: ভেক্টর ডেটা এক্সপোর্ট করুন
Export.table.toDrive({
  collection: filteredVector,       // এক্সপোর্ট করতে চাওয়া ভেক্টর ডেটা
  description: 'Exported_Vector',   // টাস্কের নাম
  folder: 'GEE_Exports',            // ফোল্ডার
  fileNamePrefix: 'vector_example', // ফাইলের নাম
  fileFormat: 'GeoJSON'             // ফাইল ফরম্যাট
});
```
