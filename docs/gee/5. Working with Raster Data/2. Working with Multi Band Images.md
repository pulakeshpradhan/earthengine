var L8 = ee.ImageCollection("LANDSAT/LC08/C02/T1_TOA"),
    roi = /* color: #d63000 */ee.Geometry.Point([73.79048080186013, 18.45022559030386]);
/***** End of imports. If edited, may not auto-convert in the playground. *****/
//Using ee.ImageCollection Constructor to call Landsat 8 Collection 2 Tier 1 TOA Reflectance
// var L8 = ee.ImageCollection("LANDSAT/LC08/C02/T1_TOA");
// print(L8.size());

// // Filter Landsat8 ImageCollection over the year 2022
var L8FilteredOverTime = L8.filterDate('2022-01-01', '2023-01-01');
print(L8FilteredOverTime.size()); 



//Further Filter L8 Image Collection over a roi and display the collection
//By Default latest Image of the ImageCollection gets displayed in Map panel
var L8Filtered = L8FilteredOverTime.filterBounds(roi);
print(L8Filtered.size()); 

Map.centerObject(roi,12);
Map.addLayer(L8Filtered,{},'L8 ImageCollection');
Map.addLayer(ee.Image('LANDSAT/LC08/C02/T1_TOA/LC08_147047_20221223'));
//Print filtered Image Collection on the console to explore metadata
print(L8Filtered); 

//Sort filtered ImageCollection by Cloud_Cover property in metadata and get the least Cloud_Cover image
var leastCloudImage = L8Filtered.sort('CLOUD_COVER').first();

print(leastCloudImage);
Map.addLayer(leastCloudImage,{},'Least Cloud Cover Image');

// //Print information about the bands in the image
// var bandNames = leastCloudImage.bandNames();
// print('Band Names:',bandNames);

// //True Colour Composite(TCC) 
Map.addLayer(leastCloudImage, {min:0, max:0.25, bands:['B4','B3','B2']},'True Color Composite');

// //False Colour Composite(FCC)
Map.addLayer(leastCloudImage, {min:0, max:0.25, bands:['B5','B4','B3']},'False Color Composite');

// //Define a variable for Visualisation parameters for ease of use
var TCCVisParams = {min:0, max:0.25, bands:['B4','B3','B2']};

//Apply a Temporal Reducer 
//Reduce Filtered ImageCollection by calculating the median of all values at each pixel 
var medianImage = L8Filtered.median();
Map.addLayer(medianImage,TCCVisParams,'Median Image');
