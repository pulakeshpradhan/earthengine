```javascript
/* Refer :https://developers.google.com/earth-engine/tutorials/tutorial_global_surface_water_01
https://global-surface-water.appspot.com/
*/
  
//Import Global Surface Water(GSW) dataset
var gsw = ee.Image('JRC/GSW1_3/GlobalSurfaceWater');

// Check the image bands and values by printing onto console and using Inspector
print(gsw);

// Define Visualisation parameters for occurrence band of gsw

var visualization = {
  min: 0.0,
  max: 100.0,
  palette: ['ffffff', 'ffbbbb', '0000ff']
};

// set you rmap view centered over Pune region 

Map.setCenter(73.8, 18.45, 11);

//Display occurrence band of gsw in the Mapview
/*Occurrence value at any pixel is the average of each of the monthly surface water occurrence
values,which normalizes against seasonal variation */

Map.addLayer(gsw.select('occurrence'), visualization, 'Occurrence');

/*--------------------------------------------------------------------------------------*/

//Create a threshold layer of water bodies using occurrence band values

var WaterMaskVisParams = {
  palette: ['white', 'blue']
};

// Create a water mask layer, and set the image mask so that non-water areas are opaque.
var occurrence = gsw.select('occurrence');
var waterMask = occurrence.gt(90).unmask(0);

Map.addLayer({
  eeObject: waterMask,
  visParams: WaterMaskVisParams,
  name: '90% occurrence water mask'
});

// /*--------------------------------------------------------------------------------------*/

var change = gsw.select("change_abs");
var changeVisParams = {
    min:-50,
    max:50,
    palette: ['red', 'black', 'limegreen']
};

Map.setCenter(78.5, 30.5, 10);  // Tehri Reservoir
//Map.setCenter(86.5, 26.0, 8);  // Kosi River
//Map.setCenter(92.5, 26.5, 8);  // Brahmaputra valley
//Map.setCenter(81.4, 16.5, 12);  // Aquaculture in Coastal AP
// Map.setCenter(78.9, 10.0, 12);  // Filling up of chain of tanks in Tamil Nadu
//Map.setCenter(75.4, 22.0, 9);  // Indirasagar and Sardar Sarovar Reservoirs

Map.addLayer(change, changeVisParams,'occurrence change intensity');


/*--------------------------------------------------------------------------------------*/

//Select the band of max-extent and display maximum extent of water bodies in the map view
var maximumExtent = gsw.select('max_extent');
// Map.addLayer(maximumExtent);

var masked = maximumExtent.updateMask(maximumExtent);
Map.addLayer(masked,{palette:'blue'}, 'Maximum extent of Water Bodies');

/*--------------------------------------------------------------------------------------*/

var transition = gsw.select('transition');

/*Define pallette of colors for 10 transition classes as defined in the documentation
of gsw layer using layer manager and then import your pallette and copy the cose 
ito the script */

var imageVisParam = {"opacity":1,"bands":["transition"],"min":0,"max":10,"palette":["ffffff","0000ff","22b14c","d1102d","99d9ea","b5e61d","e6a1aa","ff7f27","ffc90e","7f7f7f","c3c3c3"]};

Map.addLayer(transition,imageVisParam,'Transition classes');
