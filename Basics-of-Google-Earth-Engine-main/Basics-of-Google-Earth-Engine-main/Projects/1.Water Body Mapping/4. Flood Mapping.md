/**** Start of imports. If edited, may not auto-convert in the playground. ****/
var geometry = /* color: #d63000 */ee.Geometry.MultiPoint();
/***** End of imports. If edited, may not auto-convert in the playground. *****/
var date1 = ee.Date('2023-05-15');
var date2 = ee.Date('2023-07-09');
 
var polarization = 'VV';

var date1image = ee.ImageCollection('COPERNICUS/S1_GRD')
                      .filterDate(date1, date1.advance(1, 'week'))
                      .filter(ee.Filter.listContains(
                          'transmitterReceiverPolarisation', polarization))
                      .filter(ee.Filter.eq('instrumentMode', 'IW'))
                      .select(polarization)
                      .min();
                      
var date2image = ee.ImageCollection('COPERNICUS/S1_GRD')
                      .filterDate(date2, date2.advance(1, 'week'))
                      .filter(ee.Filter.listContains(
                          'transmitterReceiverPolarisation', polarization))
                      .filter(ee.Filter.eq('instrumentMode', 'IW'))
                      .select(polarization)
                      .min();                      

var nonwatermask = ee.ImageCollection('GLCF/GLS_WATER').select('water').mean().neq(2);

var date1image = date1image.updateMask(nonwatermask);
var date2image = date2image.updateMask(nonwatermask);


Map.addLayer(date1image, {min: -30, max: -5, palette: ['aqua','blue','red','brown']},"No Flood");
Map.addLayer(date2image, {min: -30, max: -5, palette: ['aqua','blue','red','brown']},"Flood");

Map.setCenter(77.2, 28.7, 12);
