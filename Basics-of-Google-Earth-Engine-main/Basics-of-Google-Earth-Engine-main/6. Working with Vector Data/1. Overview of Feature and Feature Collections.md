/*
OVERVIEW OF FEATURE AND FEATURE COLLECTION
------------------------------------------
1. Earth Engine handles vector data with the Geometry type.
2. Geometries can be interactively created using the Code Editor geometry tools.
3. Features are composed of a Geometry and a dictionary of properties. 
4. A collection of features is handled by a FeatureCollection
-------------------------------------------------------------------------------------------
*/
 
//Create a Feature from the Geometry and a dictionary of properties/attributes.

var polygon = /* color: #d63000 */ee.Geometry.Polygon(
        [[[72.42304642796515, 26.658835279422025],
          [74.53242142796515, 17.88631782279873],
          [82.96992142796515, 19.551079861031404],
          [86.48554642796515, 23.474922510185284],
          [79.45429642796515, 27.597479360670008],
          [75.58710892796515, 27.597479360670008]]]);
Map.centerObject(polygon,5);
// Map.addLayer(polygon);

var polyFeature = ee.Feature(polygon, {ID: 01, Region: 'XY'});

print(polyFeature);
//Map.addLayer(polyFeature, {}, 'feature');

/*-----------------------------------------------------------------------------------------------*/

// Make a feature and set some properties 
var feature = ee.Feature(ee.Geometry.Point([73.77247076178492, 18.439163939288836]))
  .set('type', 'Dam').set('name', 'Khadakwasla');

// Get a property from the feature.
var damName = feature.get('name');
print(damName);

// Set a new property.
feature = feature.set('river', 'Mutha');

// Overwrite the old properties with a new dictionary.
var newDict = {location: 'Pune', length :1939};
var feature = feature.set(newDict);

// Check the result.
print(feature);

/*-----------------------------------------------------------------------------------------------*/

// Search for a Feature Collection in the datacatalog and display it 
var fc = ee.FeatureCollection("FAO/GAUL_SIMPLIFIED_500m/2015/level2");
Map.setCenter(73.79, 18.45, 5);

//If no visualisation parameters are passed, it takes default values

Map.addLayer(fc, {}, 'default display'); 
// The default visualization will display the vectors with solid black lines and semi-opaque black fill

/*-----------------------------------------------------------------------------------------------
set different colors to outline/border and fillcolor 
Fillcolor- color for filling polygons and point shapes. 
First 6 digits of hexadecimal string stores color and last two digits supports opacity- '00' transparent, 'FF' opaque
width for lines and outlines for polygons and point shapes.
Refer: https://developers.google.com/earth-engine/apidocs/ee-featurecollection-style
*/

var styleParams = {
  color: '000000',
  fillColor: 'ffffff00', //Defaults to 'color' at 0.66 opacity
  width: 1.0,
};

Map.addLayer(fc.style(styleParams), {}, 'Second Level Administrative Units');

/*----------------------------------------------------------------------------------------------
change the linetype */

var styleParams = {color:'000000', fillColor:'ffffffff', width:1.0, lineType:"dotted"}; 
Map.addLayer(fc.style(styleParams), {},'Second Level Administrative Units');

/*---------------------------------------------------------------------------------------------
Check the no of features in the feature collection
*/

print(fc.size());
// print(fc); //printing to console will be aborted after collecting 5000 elements

/*---------------------------------------------------------------------------------------------
Filtering the feature collection
*/

var indiaAdm2Map = fc.filter(ee.Filter.eq('ADM0_NAME', 'India'));
print(indiaAdm2Map.size());
Map.addLayer(indiaAdm2Map.style(styleParams), {},'India Admin Map level 2');

/*---------------------------------------------------------------------------------------------
Filter by State Name 
*/

var maharashraAdm2Map = indiaAdm2Map.filter(ee.Filter.eq('ADM1_NAME', 'Maharashtra'));
var mystyle = {color:'blue',fillColor:'ffffffff', width:1}; 
Map.addLayer(maharashraAdm2Map.style(mystyle), {},'Maharashtra');
