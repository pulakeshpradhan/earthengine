// Geometry Data Types 
var point = ee.Geometry.Point([0, 0]);
print('Point:', point);
var lineString = ee.Geometry.LineString([[0, 0], [1, 1]]);
print('LineString:', lineString);
var polygon = ee.Geometry.Polygon([[[0, 0], [1, 0], [1, 1], [0, 1], [0, 0]]]);
print('Polygon:', polygon);

// Feature Data Type
var feature = ee.Feature(point, {name: 'My Point'});
print('Feature:', feature);

// FeatureCollection Data Type
var featureCollection = ee.FeatureCollection([feature]);
print('FeatureCollection:', featureCollection);

// Selecting an Image from an ImageCollection
var imageCollection = ee.ImageCollection('LANDSAT/LC08/C02/T1_TOA');
var image = imageCollection.filterDate('2018-01-01', '2018-12-31').first();
print('Image:', image);

// Creating Constant Image
var constantImage = ee.Image.constant(1);
print('Constant Image:', constantImage);
