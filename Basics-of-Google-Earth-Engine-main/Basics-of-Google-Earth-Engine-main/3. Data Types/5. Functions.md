```javascript
// Simple Function
function greet() {
    print('Hello, world!');
}
greet();  // Calls the function 

// Sum Function
function sum(a, b) {
    return a + b;
}
print('Sum:', sum(5, 3));  // Prints 8

// Index Function
function getIndex(array, element) {
    return array.indexOf(element);
}
print('Index:', getIndex([1, 2, 3, 4, 5], 3));  // Prints 2

// Average Function
function average(array) {
    var sum = array.reduce(function(a, b) { return a + b; }, 0);
    return sum / array.length;
}
print('Average:', average([1, 2, 3, 4, 5]));  // Prints 3

// Square Function
function square(x) {
    return x * x;
}
print('Square:', square(5));  // Prints 25
